from random import seed
from random import randrange
from random import random

def initialize_network(n_inputs, n_hidden, n_outputs):
	network = list()
	# weights number contains an extra one for input theta0 which is always 1 (according to coursera ml course)
	hidden_layer1 = [{'weights': [random() for i in range(n_inputs + 1)]} for i in range(n_hidden)]
	hidden_layer2 = [{'weights': [random() for i in range(n_hidden + 1)]} for i in range(n_hidden)]
	output_layer = [{'weights': [random() for i in range(n_hidden + 1)]} for i in range(n_outputs)]
	network.append(hidden_layer1)
	network.append(hidden_layer2)
	network.append(output_layer)
	return network

def net_input(weights, inputs):
	# a = sum(w * input.T) + bias
	total_input = weights[-1]
	for i in range(len(weights)-1):
		total_input += weights[i] * inputs[i]
	return total_input

def activation(total_input):
	# sigmoid or reLu or whatever
	return 1.0/ (1.0 + exp(-total_input))#此处为sigmoid激活函数，用于将网络输入映射到(-1,1)区间

#前向传播：对于每一层，都把上一次的输出作为下一层的输入，进行循环计算。
def forward_propagate(network, row):
	inputs = row
	for layer in network:
		outputs = []
		for neuron in layer:
			total_input = net_input(neuron['weights'], inputs)
			neuron['output'] = activation(total_input)
			outputs.append(neuron['output'])
		inputs = outputs
	return inputs

# sigmoid激活函数的导数实现
def transfer_derivative(output):
	# d(sigmoid(z)) = z * (1-z), standard bp algorithm
	return output * (1.0 - output)

#在反向传播计算中，cost_function函数不是必需的，根据它进行求导才是必需的。
def cost_function(expected, outputs):
	n = len(expected)
	total_error = 0.0
	for i in range(n):
		total_error += (expected[i] - outputs[i])**2
	return total_error


def backward_propagate(network, expected):#参数1，需要更新的网络模型network，它是一个list，每个item是一组参数，其中包含了权重和其它属性；参数2，期望值，包含结果的真实分类
	# error[i] = W * error[i+1] * sigmoid_derivative
	for i in reversed(range(len(network))):
		layer = network[i]
		errors = list()#从网络最后一层（输出层）开始计算，获得当前层（layer）并设置变量errors为一个list，errors将存储当前层中每个神经元的预测值的误差。这个误差是从输出层开始不断迭代累积形成的，而不是和真实目标值y的绝对误差。

		if i == len(network) - 1:#对最后一层（输出层）进行处理
			# first from output layer, just get the output difference
			# note: our difference is expected - output, some other implementing uses output-expected
			#       which causes the weight update is different (weight = weight +/- ....)
			for j in range(len(layer)):
				neuron = layer[j]
				error = -2 * (expected[j] - neuron['output'])#△w = -2·(y-output1)·input1(1-input1)·h1，将第1部分-2·(y-output1)存入errors中
				errors.append(error)
				# errors.append(expected[j] - neuron['output'])
		else:
			for j in range(len(layer)):#计算前一层的∑δ(h)·w。此处需将前一层和当前神经元相关的所有输出误差全部累加（这是全连接网络，意味着当前层的每个神经元的输出都会作用于下一层的每个神经元的输入）。
				error = 0.0
				for neuron in network[i+1]:
					# neuron['delta'] saves errors[i+1] * sigmoid_derivateive
					error += (neuron['weights'][j] * neuron['delta'])
				errors.append(error)

		for j in range(len(layer)):
			neuron = layer[j]
			neuron['delta'] = errors[j] * transfer_derivative(neuron['output'])#input1(1-input1)是sigmoid激活函数对输入值的导数。在每个神经元的delta属性中都会存储前面所计算的δ(h)。


def update_weights(network, row, learning_rate):#在back_propagate函数之后调用
	for i in range(len(network)):
		inputs = row[:-1]
		if i != 0:
			# for hidden layer and output layer
			inputs = [neuron['output'] for neuron in network[i-1]]
		for neuron in network[i]:#对该层所有神经元都进行遍历，对该神经元的每一个输入input[j]所对应的权重neuron['weights'][j]都减去△w，△w为在back_propagate函数中计算的neuron['delta']·inputs[j]。
			for j in range(len(inputs)):
				neuron['weights'][j] -= learning_rate * neuron['delta'] * inputs[j]
			# theta0 is always 1 (explained on coursera ml course)
			neuron['weights'][-1] -= learning_rate * neuron['delta']#对bias进行处理

def train_network(network, training_data, learning_rate, n_epoch, n_outputs):
	for epoch in range(n_epoch):
		sum_error = 0
		for row in training_data:
			outputs = forward_propagate(network, row)#进行前向传播获得最终输出
			# this is a trick to set the bi-class flag by using the row[-1] as class index
			# expected[0] = 1: expected to be class 1
			# expected[1] = 1: expected to be class 2
			expected = [0 for i in range(n_outputs)]
			expected[row[-1]] = 1
			# sum_error += sum([(expected[i] - outputs[i])**2 for i in range(len(expected))])
			sum_error += cost_function(expected, outputs)
			backward_propagate(network, expected)
			update_weights(network, row, learning_rate)
		print('>epoch: %d, learning rate: %.3f, error: %.3f' % (epoch, learning_rate, sum_error))


def predict(network, row):
	outputs = forward_propagate(network, row)
	return outputs.index(max(outputs))

对于任何模型训练，关键都是实现4个核心函数：  
	net_input：计算神经元的网络输入
	activation：激活函数，将神经元的网络输入映射到下一层的输入空间
	cost_function：计算误差损失
	update_weights：更新权重

前向传播（Forward Propagation）：  
指将数据输入神经网络中，每个隐藏层的神经元都接受网络输入，通过激活函数进行处理，然后进入下一层或者输出的过程。在linear_regression.md中，predict方法就是一个简单的前向传播方法。

反向传播（Back Propagation）：  
指对网络中的所有权重都计算损失函数的梯度，这个梯度会在优化算法中用来更新权值以最小化损失函数。实际上，它指代所有基于梯度下降利用链式法则（Chain Rule）来训练神经网络的算法，以帮助实现可递归循环的形式来有效地计算每一层的权重更新，直到获得期望的效果。在linear_regression.md中，把反向传播计算梯度的内容和更新权重放在了一起。

$$sigmoid(z) = φ(z) = \frac{1}{1 + e^{-z}}$$
$$\frac{\partial φ}{\partial z} = \frac{\partial(\frac{1}{1+e^{-z}})}{\partial z} = z(1 - z)$$

输出层的输入并不是原始输入值x、y，而是上一层（隐藏层）的输出。对于隐藏层来说，不能直接计算它的输出误差，因为不存在输出层的真实值y，只能通过链式法则来间接推导。也就是说，假设隐藏层有一个权重w，希望对其进行修正，那么只能从最后输出的损失函数计算出的误差倒推（反向传播）。
